# Humanクラスを定義
class Human:
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    # 年齢によって大人かどうか判定するメソッド
    def check_adult(self):
        if self.age >= 20:
            print(f"{self.name} さん（{self.age}歳）は大人です。")
        else:
            print(f"{self.name} さん（{self.age}歳）は大人ではありません。")

# Humanクラスのインスタンスを複数作成
human_list = [
    Human("さおり", 25),
    Human("たけし", 18),
    Human("マリア", 30),
    Human("ケン", 12)
]

# リストの要素分だけcheck_adultを呼び出す
for human in human_list:
    human.check_adult()
